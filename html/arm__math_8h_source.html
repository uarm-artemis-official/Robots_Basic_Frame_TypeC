569CMSIS_INLINE __STATIC_INLINE uint32_t arm_recip_q31(q31_t in, q31_t* dst,
570 q31_t* pRecipTable) {
571 q31_t out;
572 uint32_t tempVal;
573 uint32_t index, i;
574 uint32_t signBits;
575
576 if (in > 0) {
577 signBits = ((uint32_t) (__CLZ(in) - 1));
578 } else {
579 signBits = ((uint32_t) (__CLZ(-in) - 1));
580 }
581
582 /* Convert input sample to 1.31 format */
583 in = (in << signBits);
584
585 /* calculation of index for initial approximated Val */
586 index = (uint32_t) (in >> 24);
587 index = (index & INDEX_MASK);
588
589 /* 1.31 with exp 1 */
590 out = pRecipTable[index];
591
592 /* calculation of reciprocal value */
593 /* running approximation for two iterations */
594 for (i = 0U; i < 2U; i++) {
595 tempVal = (uint32_t) (((q63_t) in * out) >> 31);
596 tempVal = 0x7FFFFFFFu - tempVal;
597 /* 1.31 with exp 1 */
598 /* out = (q31_t) (((q63_t) out * tempVal) >> 30); */
599 out = clip_q63_to_q31(((q63_t) out * tempVal) >> 30);
600 }
601
602 /* write output */
603 *dst = out;
604
605 /* return num of signbits of out = 1/in value */
606 return (signBits + 1U);
607}

612CMSIS_INLINE __STATIC_INLINE uint32_t arm_recip_q15(q15_t in, q15_t* dst,
613 q15_t* pRecipTable) {
614 q15_t out = 0;
615 uint32_t tempVal = 0;
616 uint32_t index = 0, i = 0;
617 uint32_t signBits = 0;
618
619 if (in > 0) {
620 signBits = ((uint32_t) (__CLZ(in) - 17));
621 } else {
622 signBits = ((uint32_t) (__CLZ(-in) - 17));
623 }
624
625 /* Convert input sample to 1.15 format */
626 in = (in << signBits);
627
628 /* calculation of index for initial approximated Val */
629 index = (uint32_t) (in >> 8);
630 index = (index & INDEX_MASK);
631
632 /* 1.15 with exp 1 */
633 out = pRecipTable[index];
634
635 /* calculation of reciprocal value */
636 /* running approximation for two iterations */
637 for (i = 0U; i < 2U; i++) {
638 tempVal = (uint32_t) (((q31_t) in * out) >> 15);
639 tempVal = 0x7FFFu - tempVal;
640 /* 1.15 with exp 1 */
641 out = (q15_t) (((q31_t) out * tempVal) >> 14);
642 /* out = clip_q31_to_q15(((q31_t) out * tempVal) >> 14); */
643 }
644
645 /* write output */
646 *dst = out;
647
648 /* return num of signbits of out = 1/in value */
649 return (signBits + 1);
650}

5516CMSIS_INLINE __STATIC_INLINE float32_t arm_bilinear_interp_f32(
5517 const arm_bilinear_interp_instance_f32* S, float32_t X, float32_t Y) {
5518 float32_t out;
5519 float32_t f00, f01, f10, f11;
5520 float32_t* pData = S->pData;
5521 int32_t xIndex, yIndex, index;
5522 float32_t xdiff, ydiff;
5523 float32_t b1, b2, b3, b4;
5524
5525 xIndex = (int32_t) X;
5526 yIndex = (int32_t) Y;
5527
5528 /* Care taken for table outside boundary */
5529 /* Returns zero output when values are outside table boundary */
5530 if (xIndex < 0 || xIndex > (S->numRows - 1) || yIndex < 0 ||
5531 yIndex > (S->numCols - 1)) {
5532 return (0);
5533 }
5534
5535 /* Calculation of index for two nearest points in X-direction */
5536 index = (xIndex - 1) + (yIndex - 1) * S->numCols;
5537
5538 /* Read two nearest points in X-direction */
5539 f00 = pData[index];
5540 f01 = pData[index + 1];
5541
5542 /* Calculation of index for two nearest points in Y-direction */
5543 index = (xIndex - 1) + (yIndex) *S->numCols;
5544
5545 /* Read two nearest points in Y-direction */
5546 f10 = pData[index];
5547 f11 = pData[index + 1];
5548
5549 /* Calculation of intermediate values */
5550 b1 = f00;
5551 b2 = f01 - f00;
5552 b3 = f10 - f00;
5553 b4 = f00 - f01 - f10 + f11;
5554
5555 /* Calculation of fractional part in X */
5556 xdiff = X - xIndex;
5557
5558 /* Calculation of fractional part in Y */
5559 ydiff = Y - yIndex;
5560
5561 /* Calculation of bi-linear interpolated output */
5562 out = b1 + b2 * xdiff + b3 * ydiff + b4 * xdiff * ydiff;
5563
5564 /* return to application */
5565 return (out);

5580 q31_t xfract, yfract; /* X, Y fractional parts */
5581 q31_t x1, x2, y1, y2; /* Nearest output values */

5583 q31_t* pYData = S->pData; /* pointer to output table values */

5585
5586 /* Input is in 12.20 format */
5587 /* 12 bits for the table index */
5588 /* Index value calculation */
5589 rI = ((X & (q31_t) 0xFFF00000) >> 20);
5590
5591 /* Input is in 12.20 format */
5592 /* 12 bits for the table index */
5593 /* Index value calculation */
5594 cI = ((Y & (q31_t) 0xFFF00000) >> 20);
5595
5596 /* Care taken for table outside boundary */
5597 /* Returns zero output when values are outside table boundary */
5598 if (rI < 0 || rI > (S->numRows - 1) || cI < 0 || cI > (S->numCols - 1)) {
5599 return (0);
5600 }
5601
5602 /* 20 bits for the fractional part */
5603 /* shift left xfract by 11 to keep 1.31 format */
5604 xfract = (X & 0x000FFFFF) << 11U;
5605
5606 /* Read two nearest output values from the index */
5607 x1 = pYData[(rI) + (int32_t) nCols * (cI)];
5608 x2 = pYData[(rI) + (int32_t) nCols * (cI) + 1];
5609

5612 yfract = (Y & 0x000FFFFF) << 11U;

5615 y1 = pYData[(rI) + (int32_t) nCols * (cI + 1)];
5616 y2 = pYData[(rI) + (int32_t) nCols * (cI + 1) + 1];

5620 acc = ((q31_t) (((q63_t) out * (0x7FFFFFFF - yfract)) >> 32));

5623 out = ((q31_t) ((q63_t) x2 * (0x7FFFFFFF - yfract) >> 32));
5624 acc += ((q31_t) ((q63_t) out * (xfract) >> 32));
5625
5626 /* y1 * (1 - xfract) * (yfract) in 3.29(q29) and adding to acc */
5627 out = ((q31_t) ((q63_t) y1 * (0x7FFFFFFF - xfract) >> 32));
5628 acc += ((q31_t) ((q63_t) out * (yfract) >> 32));
